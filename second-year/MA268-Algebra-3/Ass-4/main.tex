\question{6}

\begin{questionbody}
Let $\R[X]$ be the $\R[T]$-module where multiplication is given by
\begin{align*}
(a_0 + a_1 & T + \dotsb + a_n T^n) \cdot f(X) \\
&= a_0 f(X) + a_1 f'(X) + a_2 f''(X) + \dotsb + a_n f^{(n)}(X)
\end{align*}
where $f^{(n)}(X)$ denotes the $n$-th derivative of $f(X)$ with respect to $X$.
\begin{enumerate}[(i)]
\item Compute $(1 + T - T^2 - 3T^5) \cdot (X + 3X^2)$.

\item Show that $\op{Span}_{\R[T]}(X^n) = \op{Span}_\R(1, X, \dotsc, X^n)$.

\item Show that $\R[X]$ is not free as an $\R[T]$-module.
\end{enumerate}
\end{questionbody}

\subsection{~} % 6.i

\begin{align*}
(1 + T - T^2 - 3T^5) \cdot (X + 3X^2) &= (X + 3X^2) + \diff X (X + 3X^2) \\
    & \qquad - \ddd 2{}X (X + 3X^2) - 3 \ddd 5{}X (X + 3X^2) \\[0.5ex]
&= (X + 3X^2) + (1 + 6X) - 6 - 3(0) \\[0.5ex]
&= -5 + 7X + 3X^2
\end{align*}

\subsection{~} % 6.ii

$\op{Span}_{\R[T]}(X^n)$ is $X^n$ multiplied by a coefficient in $\R[T]$, so
\begin{align*}
\op{Span}\nolimits_{\R[T]}(X^n) &= (a_0 + a_1 T + a_2 T^2 + \dotsb + a_n T^n) \cdot X^n \\[0.5ex]
&= a_0 X^n + a_1 \diff X X^n + a_2 \ddd 2{}X X^n + \dotsb + a_n \ddd n{}X X^n \\[0.5ex]
&= a_0 X^n + a_1 n X^{n-1} + a_2 n(n-1) X^{n-2} + \dotsb + a_n n!
\end{align*}
for some $a_i \in \R$.

Likewise,
\begin{align*}
\op{Span}\nolimits_{\R}(1, X, \dotsc, X^n) &= b_0 + b_1 X + \dotsm + b_n X^n
\end{align*}
for some $b_i \in \R$.

Both spans contain all powers of $X$ between $X^0$ and $X^n$ inclusive. By careful inspection, we see that \[
b_i = a_{n-i} \prod_{j=i+1}^{n} j
\] and conclude $\op{Span}_{\R[T]}(X^n) = \op{Span}_{\R}(1, X, \dotsc, X^n)$.

\subsection{~} % 6.iii

% Elements of $\R[X]$ can have arbitrarily large powers.
If an $\R[T]$-basis existed for $\R[X]$, it would be generated by some set $S$ containing powers of $X$. There must be some highest power of $X$ in $S$, call it $X^m$.

We would expect $\op{Span}_{\R[T]}(S) = \R[X]$ but as seen in part~\textbf{(b)}, the highest power of $X$ in $\op{Span}_{\R[T]}(S)$ is $X^m$. But of course $X^{m+1} \in \R[X]$, so $\R[X]$ cannot be spanned by a finite subset, and therefore does not have a free $\R[T]$-basis.
